import numpy as np
import pandas as pd
import os
import gen_cond_proba as gcp
import gen_proc_data as gpd
import gen_categorical_mapping as cm
import desc_stats as ds

################################## Import code and data files



main_folder="/home/mmadkour/PycharmProjects/Data"

specialty_value=pd.read_csv(main_folder+"/Supplement data/provider_meta_info.csv",usecols = ["Provider Key","Specialty"])
zip_coord = pd.read_csv(main_folder+"/Supplement data/zipcode_coordinate.csv",encoding="latin-1")
feature_list=pd.read_csv(main_folder+"/Supplement data/Idealist_feature_list.csv")
cpt_onto=pd.read_csv(main_folder+"/Supplement data/CPT HCUP.csv")

################################# Generation of conditional probabilities tables from Training cohort

risknames = ['cardiovasc_comp_comb', 'ICU_duration', 'MV_duration','neuro_delirium_comb', 'post_op_mech_proc_comb', 'sepsis_comb', 'venos_throm_comb', 'aki_3','aki_7', 'aki_all']

for riskname in risknames:
    print ("Generation conditional probabilities for", riskname)
    trainnig_data=pd.read_csv(main_folder+"/CP_Trainnig/Training_Cohort_"+riskname+".csv")
    categ_cond_proba=gcp.gen_cond_prob(trainnig_data,feature_list,trainnig_data[riskname],riskname)
    if (categ_cond_proba.isnull().any().any()):
        categ_cond_proba = categ_cond_proba[-((categ_cond_proba.Category).isnull())]
    categ_cond_proba.to_csv(main_folder+"/Lookup tables/Conditional_Probability/categorical_lookup_table_"+riskname+"_from_Training_cohort.csv")

risknames = ['cardiovasc_comp_comb', 'ICU_duration', 'MV_duration','neuro_delirium_comb', 'post_op_mech_proc_comb', 'sepsis_comb', 'venos_throm_comb', 'aki_3','aki_7', 'aki_all']

for riskname in risknames:
    print ("Data Processing for", riskname)
    categ_cond_proba = pd.read_csv(main_folder+"/Lookup tables/Conditional_Probability/categorical_lookup_table_"+riskname+"_from_Training_cohort.csv",encoding="latin-1")
    unprocessed_data=pd.read_csv(main_folder+"/Dev_Val_Cohort/Unprocessed Data/data_"+riskname+".csv")
    proc_data,numeric_lookup_table=gpd.gen_proc_data(unprocessed_data,feature_list,unprocessed_data[riskname],riskname, cpt_onto,main_folder)
    # preprocessed_data
    proc_data.to_csv(main_folder+"/Dev_Val_Cohort/Processed Data/processed_data_"+riskname+"_development_cohort.csv")
    #transformed_data
    proc_data=gpd.gen_proc_data_cpt(proc_data,feature_list,unprocessed_data[riskname],riskname, cpt_onto,main_folder)
    processed_data=cm.categorical_mapping(proc_data,categ_cond_proba,zip_coord,specialty_value,feature_list)
    fl=list(processed_data)
    myl=[]
    for item in fl:
        if item in list(feature_list["feature_name"]):
            if (feature_list.loc[feature_list["feature_name"]==item]["feature_type"]).iloc[0] in {'id','num','int','cat','bin','cpt'}:
                myl.append(item)
    myl.append('outcome')
    processed_data=processed_data[myl]
    processed_data.to_csv(main_folder+"/Dev_Val_Cohort/Transformed Data/transformed_data_"+riskname+"_development_cohort.csv")
    numeric_lookup_table.to_csv(main_folder+"/Lookup tables/Numerical/numeric_lookup_table"+riskname+".csv")
